<script>
  let transactions = JSON.parse(localStorage.getItem('finance_transactions')) || [];
  let accounts = JSON.parse(localStorage.getItem('finance_accounts')) || ['Cash', 'CIB'];
  let categoriesList = JSON.parse(localStorage.getItem('finance_categories')) || ["Transport", "Food & Drink", "Charity", "Shopping", "Work/Salary", "Adjustment", "Other"];

  let openPanels = new Set();
  let currentEditType = '';
  let currentTargetTotal = 0;
  let preDistribBalances = {};

  if(!document.getElementById('date').value) document.getElementById('date').valueAsDate = new Date();

  let lastSelectedAccount = accounts[0];
  let lastSelectedCategory = categoriesList[0];

  function saveData() {
      localStorage.setItem('finance_transactions', JSON.stringify(transactions));
      localStorage.setItem('finance_accounts', JSON.stringify(accounts));
      localStorage.setItem('finance_categories', JSON.stringify(categoriesList));
  }

  function togglePanel(dateStr) {
      const panel = document.getElementById(`details-${dateStr}`);
      const icon = document.getElementById(`icon-${dateStr}`);
      if (panel.style.display === 'none') {
          panel.style.display = 'block';
          icon.innerText = '▼';
          openPanels.add(dateStr);
      } else {
          panel.style.display = 'none';
          icon.innerText = '►';
          openPanels.delete(dateStr);
      }
  }

  function handleDynamicDropdown(selectElement, type) {
      if (selectElement.value === '_ADD_NEW_') {
          const newItem = prompt(`Enter new ${type} name:`);
          if (newItem && newItem.trim() !== '') {
              const cleanItem = newItem.trim();
              if (type === 'account' && !accounts.includes(cleanItem)) {
                  accounts.push(cleanItem);
                  lastSelectedAccount = cleanItem;
              } else if (type === 'category' && !categoriesList.includes(cleanItem)) {
                  categoriesList.push(cleanItem);
                  lastSelectedCategory = cleanItem;
              }
              saveData();
              updateUI();
          } else {
              updateUI();
          }
      } else {
          if (type === 'account') lastSelectedAccount = selectElement.value;
          if (type === 'category') lastSelectedCategory = selectElement.value;
      }
  }

  function openManageModal(type) {
      currentEditType = type;
      document.getElementById('manage-modal').style.display = 'flex';
      document.getElementById('modal-title').innerText = type === 'account' ? 'Manage Banks & Accounts' : 'Manage Categories';
      renderModalList();
  }

  function renderModalList() {
      const listContainer = document.getElementById('modal-list');
      let dataList = currentEditType === 'account' ? accounts : categoriesList;

      listContainer.innerHTML = '';
      dataList.forEach((item, index) => {
          const isSystem = (currentEditType === 'category' && item === 'Adjustment');
          const disabledAttr = isSystem ? 'disabled' : '';
          const colorAttr = isSystem ? 'color: var(--text-muted);' : '';

          listContainer.innerHTML += `
              <div class="edit-row">
                  <input type="text" id="modal-input-${index}" value="${item}" ${disabledAttr} style="${colorAttr}">
                  <button type="button" class="btn-secondary" onclick="saveListItem(${index})" ${disabledAttr}>Save</button>
                  <button type="button" class="delete-btn-modal" onclick="deleteListItem(${index})" title="Delete" ${disabledAttr}>✖</button>
              </div>
          `;
      });
  }

  function saveListItem(index) {
      const newVal = document.getElementById(`modal-input-${index}`).value.trim();
      if (!newVal) return;

      let dataList = currentEditType === 'account' ? accounts : categoriesList;
      const oldVal = dataList[index];
      dataList[index] = newVal;

      transactions.forEach(t => {
          if (currentEditType === 'account' && t.account === oldVal) t.account = newVal;
          if (currentEditType === 'category' && t.category === oldVal) t.category = newVal;
      });

      saveData();
      updateUI();

      if (currentEditType === 'account' && lastSelectedAccount === oldVal) lastSelectedAccount = newVal;
      if (currentEditType === 'category' && lastSelectedCategory === oldVal) lastSelectedCategory = newVal;
      renderModalList();
  }

  function deleteListItem(index) {
      if (confirm("Delete this? Past transactions will keep the name, but it will be removed from your options.")) {
          let dataList = currentEditType === 'account' ? accounts : categoriesList;
          dataList.splice(index, 1);
          saveData();
          updateUI();
          renderModalList();
      }
  }

  function closeModal(modalId) {
      document.getElementById(modalId).style.display = 'none';
  }

  function openTotalDistributionModal() {
      document.getElementById('total-modal').style.display = 'flex';
      document.getElementById('total-step-1').style.display = 'block';
      document.getElementById('total-step-2').style.display = 'none';

      const totalInput = document.getElementById('new-total-input');
      totalInput.value = '';
      setTimeout(() => totalInput.focus(), 100);

      // Get snapshot of current balances
      preDistribBalances = {};
      accounts.forEach(acc => preDistribBalances[acc] = 0);
      transactions.forEach(t => {
          if (accounts.includes(t.account)) {
              if(t.type === 'income') preDistribBalances[t.account] += t.amount;
              else preDistribBalances[t.account] -= t.amount;
          }
      });
  }

  function goToDistribution() {
      const val = parseFloat(document.getElementById('new-total-input').value);
      if (isNaN(val)) return;

      currentTargetTotal = val;
      document.getElementById('total-step-1').style.display = 'none';
      document.getElementById('total-step-2').style.display = 'block';
      document.getElementById('display-new-total').innerText = val.toLocaleString();

      const container = document.getElementById('distribution-inputs');
      container.innerHTML = '';

      accounts.forEach(acc => {
          // UPDATED: Changed step="0.01" to step="10" for the distribution inputs
          container.innerHTML += `
              <div class="form-group" style="margin-bottom: 10px; display: flex; flex-direction: row; align-items: center; justify-content: space-between;">
                  <label style="flex: 1; font-size: 1.1em; font-weight: bold;">${acc}</label>
                  <input type="number" step="10" class="distrib-input" data-account="${acc}" value="${preDistribBalances[acc]}" oninput="updateRemaining()" style="flex: 2; text-align: right; font-size: 1.1em;">
              </div>
          `;
      });

      updateRemaining();
  }

  function updateRemaining() {
      const inputs = document.querySelectorAll('.distrib-input');
      let currentSum = 0;
      inputs.forEach(input => currentSum += (parseFloat(input.value) || 0));

      const remaining = currentTargetTotal - currentSum;
      const remainingEl = document.getElementById('left-to-assign');

      if (Math.abs(remaining) < 0.01) {
          remainingEl.innerText = "0";
          remainingEl.className = 'text-teal';
          document.getElementById('save-distribution-btn').disabled = false;
      } else {
          remainingEl.innerText = remaining > 0 ? "+" + remaining.toLocaleString() : remaining.toLocaleString();
          remainingEl.className = 'text-pink';
          document.getElementById('save-distribution-btn').disabled = true;
      }
  }

  function saveTotalDistribution() {
      const inputs = document.querySelectorAll('.distrib-input');
      const today = new Date().toISOString().split('T')[0];

      inputs.forEach(input => {
          const acc = input.getAttribute('data-account');
          const newVal = parseFloat(input.value) || 0;
          const diff = newVal - preDistribBalances[acc];

          if (Math.abs(diff) > 0.01) {
              transactions.push({
                  id: Date.now() + Math.floor(Math.random() * 1000),
                  date: today,
                  type: diff > 0 ? 'income' : 'expense',
                  amount: Math.abs(diff),
                  account: acc,
                  category: 'Adjustment',
                  description: 'Total Balance Re-distribution'
              });
          }
      });

      saveData();
      updateUI();
      closeModal('total-modal');
  }

  function adjustBalance(accountName, currentBalance) {
      const input = prompt(`Current ${accountName} balance is EGP ${currentBalance}.\nEnter your ACTUAL current balance to adjust:`, currentBalance);

      if (input !== null && input !== "") {
          const newBalance = parseFloat(input);
          if (!isNaN(newBalance) && newBalance !== currentBalance) {
              const difference = newBalance - currentBalance;
              transactions.push({
                  id: Date.now(),
                  date: new Date().toISOString().split('T')[0],
                  type: difference > 0 ? 'income' : 'expense',
                  amount: Math.abs(difference),
                  account: accountName,
                  category: 'Adjustment',
                  description: 'Manual Balance Correction'
              });
              saveData();
              updateUI();
          }
      }
  }

  document.getElementById('transaction-form').addEventListener('submit', function(e) {
      e.preventDefault();

      const dateVal = document.getElementById('date').value;
      const amountVal = parseFloat(document.getElementById('amount').value);
      const descVal = document.getElementById('description').value;

      if(!dateVal || isNaN(amountVal) || !descVal) {
          alert("Please fill out the form properly.");
          return;
      }

      transactions.push({
          id: Date.now(),
          date: dateVal,
          type: document.getElementById('type').value,
          amount: amountVal,
          account: document.getElementById('account').value,
          category: document.getElementById('category').value,
          description: descVal
      });

      saveData();
      updateUI();

      document.getElementById('amount').value = '';
      document.getElementById('description').value = '';
      document.getElementById('amount').focus();
  });

  function deleteTransaction(id) {
      transactions = transactions.filter(t => t.id !== id);
      saveData();
      updateUI();
  }

  function makeEditable(element, id, field, type) {
      if (element.querySelector('.inline-edit-input')) return;

      const currentVal = transactions.find(t => t.id === id)[field];
      let inputHTML = '';

      if (type === 'select-account') {
          const options = accounts.map(a => `<option value="${a}" ${a === currentVal ? 'selected' : ''}>${a}</option>`).join('');
          inputHTML = `<select class="inline-edit-input">${options}</select>`;
      } else if (type === 'select-category') {
          const options = categoriesList.map(c => `<option value="${c}" ${c === currentVal ? 'selected' : ''}>${c}</option>`).join('');
          inputHTML = `<select class="inline-edit-input">${options}</select>`;
      } else if (type === 'number') {
          inputHTML = `<input type="number" step="10" class="inline-edit-input" value="${currentVal}">`;
      } else if (type === 'date') {
          inputHTML = `<input type="date" class="inline-edit-input" value="${currentVal}">`;
      } else {
          inputHTML = `<input type="text" class="inline-edit-input" value="${currentVal}">`;
      }

      element.innerHTML = inputHTML;

      const inputEl = element.querySelector('.inline-edit-input');
      inputEl.focus();

      const saveEdit = () => {
          const newVal = type === 'number' ? parseFloat(inputEl.value) : inputEl.value;
          if (newVal || newVal === 0) {
              const tIndex = transactions.findIndex(t => t.id === id);
              transactions[tIndex][field] = newVal;
              saveData();
          }
          updateUI();
      };

      inputEl.addEventListener('blur', saveEdit);
      inputEl.addEventListener('keypress', function(e) { if (e.key === 'Enter') saveEdit(); });
  }

  function updateUI() {
      const accSelect = document.getElementById('account');
      accSelect.innerHTML = accounts.map(acc => `<option value="${acc}">${acc}</option>`).join('') + `<option value="_ADD_NEW_" style="color:var(--accent-teal); font-weight:bold;">+ Add New Bank...</option>`;
      accSelect.value = accounts.includes(lastSelectedAccount) ? lastSelectedAccount : accounts[0];

      const catSelect = document.getElementById('category');
      catSelect.innerHTML = categoriesList.map(cat => `<option value="${cat}">${cat}</option>`).join('') + `<option value="_ADD_NEW_" style="color:var(--accent-teal); font-weight:bold;">+ Add New Category...</option>`;
      catSelect.value = categoriesList.includes(lastSelectedCategory) ? lastSelectedCategory : categoriesList[0];

      let totalBal = 0;
      let accountBalances = {};
      accounts.forEach(acc => accountBalances[acc] = 0);

      const chronoTransactions = [...transactions].sort((a, b) => new Date(a.date) - new Date(b.date) || a.id - b.id);

      let dailyData = {};
      let runningTotal = 0;

      chronoTransactions.forEach(t => {
          if (!accountBalances[t.account]) accountBalances[t.account] = 0;
          if (!dailyData[t.date]) dailyData[t.date] = { spent: 0, earned: 0, transactions: [], endOfDayBal: 0 };

          if (t.type === 'income') {
              totalBal += t.amount;
              accountBalances[t.account] += t.amount;
              runningTotal += t.amount;
              dailyData[t.date].earned += t.amount;
          } else {
              totalBal -= t.amount;
              accountBalances[t.account] -= t.amount;
              runningTotal -= t.amount;
              if(t.category !== 'Adjustment') dailyData[t.date].spent += t.amount;
          }

          dailyData[t.date].endOfDayBal = runningTotal;
          dailyData[t.date].transactions.push(t);
      });

      const dashboard = document.getElementById('dynamic-dashboard');
      dashboard.innerHTML = `
          <div class="card">
              <span class="stat-title">Total Balance</span>
              <div class="stat-value text-teal" onclick="openTotalDistributionModal()">
                  EGP ${totalBal.toLocaleString()} <span class="edit-hint">✎ Edit</span>
              </div>
          </div>
      `;

      accounts.forEach(acc => {
          dashboard.innerHTML += `
              <div class="card">
                  <span class="stat-title">${acc} Balance</span>
                  <div class="stat-value" onclick="adjustBalance('${acc}', ${accountBalances[acc]})">
                      EGP ${accountBalances[acc].toLocaleString()} <span class="edit-hint">✎ Edit</span>
                  </div>
              </div>
          `;
      });

      const historyContainer = document.getElementById('history-container');
      historyContainer.innerHTML = '';
      const sortedDates = Object.keys(dailyData).sort((a, b) => new Date(b) - new Date(a));

      sortedDates.forEach(date => {
          const data = dailyData[date];
          const displayDate = date.split('-').reverse().join('/');

          const isPanelOpen = openPanels.has(date);
          const displayStyle = isPanelOpen ? 'block' : 'none';
          const iconStr = isPanelOpen ? '▼' : '►';

          let html = `
          <div class="daily-group">
              <div class="daily-header">
                  <span class="clickable-date" onclick="togglePanel('${date}')">
                      <span id="icon-${date}">${iconStr}</span> ${displayDate}
                  </span>
                  <div class="daily-stats">
                      <span class="text-pink">Spent: EGP ${data.spent.toLocaleString()}</span>
                      <span class="text-teal">In: EGP ${data.earned.toLocaleString()}</span>
                      <span style="color:var(--text-muted)">| Total Bal: EGP ${data.endOfDayBal.toLocaleString()}</span>
                  </div>
              </div>
              <div id="details-${date}" style="display: ${displayStyle};">
                  <table class="daily-table"><tbody>`;

          data.transactions.sort((a, b) => b.id - a.id).forEach(t => {
              const typeClass = t.type === 'income' ? 'badge-income' : 'badge-expense';
              const typeSymbol = t.type === 'income' ? '+' : '-';

              html += `
                  <tr>
                      <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'description', 'text')" width="40%">${t.description}</td>
                      <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'category', 'select-category')" width="20%" style="color: var(--accent-orange); font-size: 0.85em;">${t.category}</td>
                      <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'account', 'select-account')" width="20%" style="font-weight:bold;">${t.account}</td>
                      <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'amount', 'number')" width="15%"><span class="type-badge ${typeClass}">${typeSymbol}${t.amount}</span></td>
                      <td width="5%"><button class="delete-btn" onclick="deleteTransaction(${t.id})">X</button></td>
                  </tr>`;
          });
          html += `</tbody></table></div></div>`;
          historyContainer.innerHTML += html;
      });

      if(sortedDates.length === 0) {
          historyContainer.innerHTML = `<p style="color: var(--text-muted); text-align: center; padding: 20px;">No transactions logged yet.</p>`;
      }

      const historyBody = document.getElementById('history-body');
      historyBody.innerHTML = '';
      const sortedTransactions = [...transactions].sort((a, b) => b.id - a.id);

      sortedTransactions.forEach(t => {
          const row = document.createElement('tr');
          const typeClass = t.type === 'income' ? 'badge-income' : 'badge-expense';
          const typeSymbol = t.type === 'income' ? '+' : '-';
          const displayDate = t.date ? t.date.split('-').reverse().join('/') : 'N/A';

          row.innerHTML = `
              <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'date', 'date')" style="color: var(--text-muted); font-size: 0.9em;">${displayDate}</td>
              <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'description', 'text')">${t.description}</td>
              <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'category', 'select-category')"><span style="font-size: 0.85em; color: var(--accent-orange);">${t.category}</span></td>
              <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'account', 'select-account')"><span style="font-weight: bold;">${t.account}</span></td>
              <td class="editable" ondblclick="makeEditable(this, ${t.id}, 'amount', 'number')"><span class="type-badge ${typeClass}">${typeSymbol}${t.amount}</span></td>
              <td><button class="delete-btn" onclick="deleteTransaction(${t.id})">X</button></td>
          `;
          historyBody.appendChild(row);
      });
  }

  updateUI();
